00267 
00268 bool SymMatrix3x3::invert()
00269 {
00270  double det, pos, neg, t, out[6];
00271 
00272  pos = neg = 0.0;
00273  t =  M[0]*M[2]*M[5]; ((t >= 0.0)?(pos):(neg)) += t;
00274  t =  M[1]*M[4]*M[3]; ((t >= 0.0)?(pos):(neg)) += t;
00275  t =  M[3]*M[1]*M[4]; ((t >= 0.0)?(pos):(neg)) += t;
00276  t = -M[3]*M[2]*M[3]; ((t >= 0.0)?(pos):(neg)) += t;
00277  t = -M[1]*M[1]*M[5]; ((t >= 0.0)?(pos):(neg)) += t;
00278  t = -M[0]*M[4]*M[4]; ((t >= 0.0)?(pos):(neg)) += t;
00279  det = pos+neg;
00280 
00281  t = det/(pos-neg);
00282  if (FABS(t) >= 1.0e-15)
00283  {
00284   out[0] =  (M[2] * M[5] - M[4] * M[4]) /det;
00285   out[1] = -(M[1] * M[5] - M[4] * M[3]) /det;
00286   out[2] =  (M[0] * M[5] - M[3] * M[3]) /det;
00287   out[3] =  (M[1] * M[4] - M[2] * M[3]) /det;
00288   out[4] = -(M[0] * M[4] - M[1] * M[3]) /det;
00289   out[5] =  (M[0] * M[2] - M[1] * M[1]) /det;
00290   M[0] = out[0]; M[1] = out[1]; M[2] = out[2];
00291   M[3] = out[3]; M[4] = out[4]; M[5] = out[5];

00299 // Compute eigenvalues and eigenvectors of the matrix (JACOBI method).
00300 // The calling function is responsible of verifying that the matrix
00301 // is diagonalizable.
00302 //
00303 // This routine was inspired from a software to estimate curvature
00304 // tensors developed at INRIA. Visit the following link for details:
00305 // http://www-sop.inria.fr/geometrica/team/Pierre.Alliez/demos/curvature/
00306 //
00307 // This version has been slightly optimized.
00308 //
00309 
00310 void SymMatrix3x3::diagonalize(double *eigen_val, double *eigen_vec) const
00311 {
00312  static const double EPS = 0.00001;
00313  static const double cos_pi_4 = 0.70710678;
00314  static const int MAX_ITER = 100;
00315  double a_P[6], v_P[9];
00316  double *a = a_P, *v = v_P;
00317  double a_norm,a_normEPS,thr,thr_nn;
00318  int nb_iter = 0;
00319  int i,j,k,ij,jj,ik,l,m,lm,mq,lq,ll,mm,imv,im,iq,ilv,il;
00320  int index_P[3];
00321  int *index = index_P;
00322  double a_ij,a_lm,a_ll,a_mm,a_im,a_il,a_lm_2,v_ilv,v_imv,x;
00323  double sinx,sinx_2,cosx,cosx_2,sincos,delta;
00324 
00325  a[0] = M[0]; a[1] = M[1]; a[2] = M[2];
00326  a[3] = M[3]; a[4] = M[4]; a[5] = M[5];
00327  a--;
00328     
00329  // Step 2 : Init diagonalization matrix as the unit matrix
00330     
00331  for (ij=0, i=0; i<3; i++) for (j=0; j<3; j++) v[ij++] = (i==j)?(1.0):(0.0); 
00332  v--;
00333     
00334  // Step 3 : compute the weight of the non diagonal terms
00335 
00336  a_norm = 0.0;
00337  for (i=1, ij=1; i<=3; i++) for (j=1; j<=i; j++, ij++) if( i!=j ) {a_ij = a[ij]; a_norm += a_ij*a_ij;}
00338     
00339  if( a_norm != 0.0 )
00340  {   
00341    a_normEPS = a_norm*EPS;
00342    thr   = a_norm  ;
00343   
00344    // Step 4 : rotations
00345    while (thr > a_normEPS && nb_iter < MAX_ITER)
00346    { 
00347    nb_iter++;
00348    thr_nn = thr / 6;
00349     
00350    for (l=1; l<3; l++)
00351    for (m=l+1; m<=3; m++)
00352    {     
00353      // compute sinx and cosx 
00354         
00355      lq = (l*l-l)/2; mq = (m*m-m)/2;    
00356      lm = l+mq; a_lm = a[lm];
00357      a_lm_2 = a_lm*a_lm;
00358         
00359      if( a_lm_2 < thr_nn ) continue;
00360         
00361      ll = l+lq; mm = m+mq;
00362      a_ll = a[ll]; a_mm = a[mm];  
00363      delta = a_ll - a_mm;
00364         
00365      if (delta==0.0) {sinx = -cos_pi_4; cosx = cos_pi_4;}
00366      else {x = -atan( (a_lm+a_lm) / delta )/2.0; sinx=sin(x); cosx=cos(x);}
00367 
00368      sinx_2  = sinx*sinx;
00369      cosx_2  = cosx*cosx;
00370      sincos  = sinx*cosx;
00371         
00372      // rotate L and M columns 
00373     
00374      ilv = 3*(l-1); imv = 3*(m-1);
00375         
00376      for( i=1; i<=3;i++ )
00377      {
00378      if( (i!=l) && (i!=m) )
00379      {
00380      iq = (i*i-i)/2;
00381      im = (i<m)?(i+mq):(m+iq);   
00382      a_im = a[im];
00383      il = (i<l)?(i+lq):(l+iq);   
00384      a_il = a[il];
00385      a[il] =  a_il*cosx - a_im*sinx;
00386      a[im] =  a_il*sinx + a_im*cosx;
00387      }
00388         
00389      ilv++; imv++;  
00390      v_ilv = v[ilv]; v_imv = v[imv];
00391      v[ilv] = cosx*v_ilv - sinx*v_imv;
00392      v[imv] = sinx*v_ilv + cosx*v_imv;
00393      } 
00394         
00395      x = a_lm*sincos; x+=x;
00396         
00397      a[ll] =  a_ll*cosx_2 + a_mm*sinx_2 - x;
00398      a[mm] =  a_ll*sinx_2 + a_mm*cosx_2 + x;
00399      a[lm] =  0.0;
00400      thr = FABS( thr - a_lm_2 );
00401    }
00402    }   
00403  }
00404     
00405  // Step 5: index conversion and copy eigen values 
00406     
00407  // back from Fortran to C++
00408  a++;
00409     
00410  for( i=0; i<3; i++ ) { k = i + (i*(i+1))/2; eigen_val[i] = a[k];}
00411       
00412  // Step 6: sort the eigen values and eigen vectors 
00413     
00414  for( i=0; i<3; i++ ) index[i] = i;
00415     
00416  for( i=0; i<(3-1); i++ ) {
00417    x = eigen_val[i];
00418    k = i;
00419     
00420    for( j=i+1; j<3; j++ ) if( x < eigen_val[j] ) { k = j; x = eigen_val[j];}
00421     
00422    eigen_val[k] = eigen_val[i];
00423    eigen_val[i] = x;
00424     
00425    jj   = index[k];
00426    index[k] = index[i];
00427    index[i] = jj;
00428  }
00429 
00430  // Step 7: save the eigen vectors 
00431   
00432  v++; // back from Fortran to to C++
00433     
00434  ij = 0;
00435  for( k=0; k<3; k++ ) for(ik = index[k]*3, i=0; i<3; i++ ) eigen_vec[ij++] = v[ik++];
00436 }

00440 
00441 void SymMatrix3x3::getEigenvalues(double *L1, double *L2, double *L3) const
00442 {
00443  double a11 = M[0], a12 = M[1], a22 = M[2], a13 = M[3], a23 = M[4], a33 = M[5];
00444  double c0 = (a11*a22*a33)+(2.0*a12*a13*a23)-(a11*a23*a23)-(a22*a13*a13)-(a33*a12*a12); // Mat. Determinant
00445  double c1 = a11*a22-a12*a12+a11*a33-a13*a13+a22*a33-a23*a23;
00446  double c2 = a11+a22+a33;
00447  double a = (3*c1-c2*c2)/3.0;
00448  double b = (9*c1*c2-2*c2*c2*c2-27*c0)/27.0;
00449  double Q = ((b*b)/4.0)+((a*a*a)/27.0);
00450 
00451  if (Q>1.0e-12) {*L1 = *L2 = *L3 = a11; return;} // Evecs = (1,0,0), (0,1,0), (0,0,1)
00452 
00453  double l1, l2, l3; // Eigenvalues to be computed
00454 
00455  if (Q>=0) {double p=(b>0)?(pow(b/2, 1.0/3.0)):(0); l1=l2=((c2/3.0)+p); l3=((c2/3.0)-2.0*p);}
00456  else
00457  {
00458   double t = atan2(sqrt(-Q), -b/2.0)/3.0, r = pow(((b*b)/4.0)-Q, 1.0/6.0);
00459   double cos_t = cos(t), sin_t = sin(t);
00460   const double sq3 = sqrt(3.0);
00461   l1 = l2 = l3 = (c2/3.0);
00462   l1 += (2*r*cos_t);
00463   l2 -= r*(cos_t+sq3*sin_t);
00464   l3 -= r*(cos_t-sq3*sin_t);
00465  }
00466 
00467  if (l1<=l2 && l1<=l3) {*L1=l1; *L2=(l2<l3)?(l2):(l3); *L3=(l2<l3)?(l3):(l2);}
00468  else if (l2<=l1 && l2<=l3) {*L1=l2; *L2=(l1<l3)?(l1):(l3); *L3=(l1<l3)?(l3):(l1);}
00469  else {*L1=l3; *L2=(l1<l2)?(l1):(l2); *L3=(l1<l2)?(l2):(l1);}
00470 }

00476 {

00480 
00481  if (l==l3 && l==l2) {*x = 1; *y = *z = 0; return;}
00482 
00483  a11-=l; a22-=l; a33-=l;
00484  double u11 = a22*a33-a23*a23, u12 = a13*a23-a12*a33, u13 = a12*a23-a13*a22;
00485  double u22 = a11*a33-a13*a13, u23 = a12*a13-a23*a11, u33 = a11*a22-a12*a12;

00644 // Invert a symmetric 4x4 matrix using L*D*L^T decomposition.
00645 // The calling function is responsible of verifying that the matrix
00646 // is positive definite.

00649 bool SymMatrix4x4::invert()
00650 {

00652 
00653  double d00 = 1.0/a2; 
00654  double L10 = ab;
00655  double l10 = ab*d00;
00656  double L20 = ac;
00657  double l20 = ac*d00;
00658  double L30 = ad;
00659  double l30 = ad*d00;
00660  double d11 = b2-(L10*l10);
00661 
00662  if (d11 <= 0) return false; else d11 = 1.0/d11;
00663 
00664  double L21 = (bc-(L10*l20));
00665  double l21 = L21*d11;
00666  double L31 = (bd-(L10*l30));
00667  double l31 = L31*d11;
00668  double d22 = c2-(L20*l20)-(L21*l21);
00669 
00670  if (d22 <= 0) return false; else d22 = 1.0/d22;
00671 
00672  double L32 = (cd-(L20*l30)-(L21*l31));
00673  double l32 = L32*d22;
00674  double d33 = d2-(L30*l30)-(L31*l31)-(L32*l32);
00675 
00676  if (d33 <= 0) return false; else d33 = 1.0/d33;
00677 
00678  L20 = l10*l21-l20;
00679  L31 = l21*l32-l31;
00680  L30 = l32*l20-L31*l10-l30;
00681 
00682  a2 = d00+(-l10)*((-l10)*d11)+L20*(L20*d22)+L30*(L30*d33);
00683  ab = ((-l10)*d11)+L20*((-l21)*d22)+L30*(L31*d33);
